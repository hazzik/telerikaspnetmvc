This example shows how Telerik Editor for ASP.NET MVC is being validated through the built-in server-side validation capabilities of ASP.NET MVC. 


<%= Html.ValidationSummary() %>
<% using (Html.BeginForm("Action", "Controller")) { %>
    <%= Html.Telerik().Editor()
              .Name("Notes")
    %>
    <%= Html.ValidationMessage(e => e.Notes, "*") %>
    <input type="submit" value="Post" />
<% } %>
